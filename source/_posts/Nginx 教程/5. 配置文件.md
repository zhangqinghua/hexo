```java
 doSomething().then(function () {
    return doSomethingElse();
    // Promise.resolve(what doSomethingElse returns)
  });

  doSomething().then(function () {
    doSomethingElse();
    // Promise.resolve(undefined)
  });

  doSomething().then(
    doSomethingElse()
    // this step will be ignored
    // as then should contain a function definition
```

```conf
server {
    listen 80;
    server_name www.china-note.com;
    return 301 https://$host$request_uri;
    index index.html index.htm index.jsp;

    access_log /data/wwwlogs/www.china-note.com_nginx.log combined;
}

server {
    listen 443 ssl;
    server_name www.china-note.com;
    index index.html index.htm index.jsp;

    ssl on;
    ssl_certificate     /usr/local/nginx/conf/ssl/www.china-note.com/www.china-note.com.pem;
    ssl_certificate_key /usr/local/nginx/conf/ssl/www.china-note.com/www.china-note.com.key;
    ssl_protocols TLSv1 TLSv1.1 TLSv1.2;
    ssl_prefer_server_ciphers on;
    ssl_ciphers 'EECDH+AESGCM:EDH+AESGCM:AES256+EECDH:AES256+EDH';

    access_log /data/wwwlogs/www.china-note.com_nginx.log combined;

    location ^~ /admin {
        alias /data/deploy/Web/newProjects/overseasProperty/overseasPropertySystem/dist;
        try_files $uri $uri/ /admin/index.html;
    }

    location ^~ /api {
        proxy_pass http://localhost:60834;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header X-nginx-real-ip $remote_addr;
        proxy_set_header x-forwarded-for $proxy_add_x_forwarded_for;
    }
}
```
